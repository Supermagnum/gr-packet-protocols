26// KISS Command Types

28#define KISS_CMD_TXDELAY 0x01 // TX Delay
29#define KISS_CMD_P 0x02 // Persistence
30#define KISS_CMD_SLOTTIME 0x03 // Slot Time
31#define KISS_CMD_TXTAIL 0x04 // TX Tail
32#define KISS_CMD_FULLDUP 0x05 // Full Duplex
33#define KISS_CMD_SETHARD 0x06 // Set Hardware
34#define KISS_CMD_RETURN 0xFF // Return

36// KISS Frame Structure
37typedef struct {
38 uint8_t* data; // Frame data
39 uint16_t length; // Frame length
40 uint8_t command; // KISS command
41 uint8_t port; // Port number
42 bool escaped; // Escape sequence flag
43} kiss_frame_t;
