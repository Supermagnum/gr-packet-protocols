26// AX.25 Frame Types

28#define AX25_FRAME_S 0x01 // Supervisory frame
29#define AX25_FRAME_U 0x03 // Unnumbered frame

31// AX.25 Control Field Types
32#define AX25_CTRL_I 0x00 // Information
33#define AX25_CTRL_RR 0x01 // Receive Ready
34#define AX25_CTRL_RNR 0x05 // Receive Not Ready
35#define AX25_CTRL_REJ 0x09 // Reject
36#define AX25_CTRL_SABM 0x2F // Set Asynchronous Balanced Mode
37#define AX25_CTRL_SABME 0x6F // Set Asynchronous Balanced Mode Extended
38#define AX25_CTRL_DISC 0x43 // Disconnect
39#define AX25_CTRL_DM 0x0F // Disconnected Mode
40#define AX25_CTRL_UA 0x63 // Unnumbered Acknowledge
41#define AX25_CTRL_FRMR 0x87 // Frame Reject
42#define AX25_CTRL_UI 0x03 // Unnumbered Information

44// AX.25 PID Types

46#define AX25_PID_IP 0xCC // Internet Protocol
47#define AX25_PID_ARP 0xCD // Address Resolution Protocol

51// AX.25 Address Structure
52typedef struct {
53 uint8_t callsign[6]; // 6-byte callsign
54 uint8_t ssid; // SSID (4 bits) + C/R, H bits
55 bool command; // Command/Response bit
56 bool has_been_repeated; // Has Been Repeated bit
57} ax25_address_t;

59// AX.25 Frame Structure
60typedef struct {
61 ax25_address_t addresses[AX25_MAX_ADDRS]; // Address fields
62 uint8_t num_addresses; // Number of addresses
63 uint8_t control; // Control field
64 uint8_t pid; // Protocol ID
65 uint8_t info[AX25_MAX_INFO]; // Information field
66 uint16_t info_length; // Information length
67 uint16_t fcs; // Frame Check Sequence
68 bool valid; // Frame validity
69} ax25_frame_t;
