37// Reed-Solomon codec control block
38struct fx25_rs {
39 unsigned int mm; // Bits per symbol
40 unsigned int nn; // Symbols per block (= (1<<mm)-1)
41 unsigned char* alpha_to; // log lookup table
42 unsigned char* index_of; // Antilog lookup table
43 unsigned char* genpoly; // Generator polynomial
44 unsigned int nroots; // Number of generator roots = number of parity symbols
45 unsigned char fcr; // First consecutive root, index form
46 unsigned char prim; // Primitive element, index form
47 unsigned char iprim; // prim-th root of 1, index form
48};
